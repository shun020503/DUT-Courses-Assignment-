int sensor = A0;
int value;
int light = 13;
void setup() {
  Serial.begin(9600);
  pinMode(light, OUTPUT);
  digitalWrite(light,LOW);
  pinMode(sensor, INPUT);
}

void loop() {
  value = analogRead(sensor);
  if (value > 500) digitalWrite(light, LOW);
  else digitalWrite(light,HIGH);
  Serial.print("Value: ");
  Serial.println(value);
  delay(200);
}
